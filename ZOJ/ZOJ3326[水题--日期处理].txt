bool IsPrime(int x)
{
     if(x <= 1) return false;
     int i;
     for(i=2;i*i<=x;i++)
        if(x % i == 0) return false;
     return true;
}
bool IsLeap(int x)
{
     if((x%400==0) || (x%4==0 && x%100!=0))
         return true;
     return false;
}
int month_prime(int month, int year)
{
    int ans=0;
    if(month <= 2) return 0;
    else if(month == 3) ans=9;
    else if(month == 4) ans=20;
    else if(month == 5) ans=20;
    else if(month == 6) ans=31;
    else if(month == 7) ans=31;
    else if(month == 8) ans=42;
    else if(month == 9) ans=42;
    else if(month == 10) ans=42;
    else if(month == 11) ans=42;
    else if(month == 12) ans=52;
    if(IsLeap(year))
       ans++;
    return ans;
}
int day_prime(int day, int month)
{
    if(month!=2 && month!=3 && month!=5 && month!=7 && month!=11) return 0;
    if(day < 2) return 0;
    if(day < 3) return 1;
    if(day < 5) return 2;
    if(day < 7) return 3;
    if(day < 11) return 4;
    if(day < 13) return 5;
    if(day < 17) return 6;
    if(day < 19) return 7;
    if(day < 23) return 8;
    if(day < 29) return 9;
    if(day < 31) return 10;
    return 11;
}
