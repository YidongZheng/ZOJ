#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
    char a[1000],b[1000];
    int i,j,k;
    int AB[1000][1000];
    while(scanf("%s%s",a,b) != EOF)
    {
           //printf("%sºÍ%s\n",a,b);
           //printf("%d\n",strlen(a));
           //printf("%d\n",strlen(b));
           for(i=0;i<strlen(a);i++)
             for(j=0;j<strlen(b);j++)
             {
                 
                 if(i+j==0)
                 {
                   if(a[0]==b[0])
                      AB[0][0]=1;
                   else
                      AB[0][0]=0;
                 }
                 else if(i==0)
                 {
                      if(a[0]==b[j])
                        AB[0][j]=1;
                      else
                        AB[0][j]=AB[0][j-1];
                 }
                 else if(j==0)
                 {
                      if(a[i]==b[0])
                        AB[i][0]=1;
                      else
                        AB[i][0]=AB[i-1][0];
                 }
                 else
                 {
                      for(k=i;k>=0;k--)
                         if(a[k]==b[j])
                            break;
                      if(k==0)
                         AB[i][j]=AB[i][j-1]>1?AB[i][j-1]:1;
                      else if(k==-1)
                         AB[i][j]=AB[i][j-1];
                      else
                         AB[i][j]=AB[i][j-1]>(AB[k-1][j-1]+1)?AB[i][j-1]:(AB[k-1][j-1]+1);
                 }
             }
           printf("%d\n",AB[i-1][j-1]);

    }
    return 0;
}
